from mpl_toolkits.mplot3d import Axes3D
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from pylab import *

Max = 50

filename4 = '/home/csachs/data/AgAu/dimer/50nm/2nm/lam600/E1_field.csv_0'
filename5 = '/home/csachs/data/AgAu/dimer/50nm/2nm/lam600/E2_field.csv_0'
data4 = pd.read_csv(filename4, header=None, skiprows=2, sep="\s+").values
data5 = pd.read_csv(filename5, header=None, skiprows=2, sep="\s+").values
XL = data4[:,0]
YL = data4[:,1]
ZL = data4[:,2]
CL = data4[:,6]

fig = plt.figure(figsize=(10,5))
ax = plt.axes(projection='3d')

ax.scatter3D(XL, YL, ZL, c=CL, cmap='copper', vmin=0, vmax=Max)
ax.view_init(0, 0) #rotates plot

# Set title and axis labels
ax.set_title('50nm AgAu - 2nm Spacing - $\lambda$=600nm - L-Mode');
ax.set_xticklabels([])
ax.set_ylabel('Distance ($\mu$m)')
ax.yaxis.labelpad = 20
ax.set_zlabel('Distance ($\mu$m)')

# make the panes transparent
ax.xaxis.set_pane_color((1.0, 1.0, 1.0, 0.0))
ax.yaxis.set_pane_color((1.0, 1.0, 1.0, 0.0))
ax.zaxis.set_pane_color((1.0, 1.0, 1.0, 0.0))
# make the grid lines transparent
ax.xaxis._axinfo["grid"]['color'] =  (1,1,1,0)
ax.yaxis._axinfo["grid"]['color'] =  (1,1,1,0)
ax.zaxis._axinfo["grid"]['color'] =  (1,1,1,0)

plt.show()
